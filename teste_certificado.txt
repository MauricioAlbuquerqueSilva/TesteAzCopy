
openssl req -x509 -newkey rsa:2048 -keyout azure_sp_private_key.pem -out azure_sp_cert.pem -days 365 -nodes
openssl x509 -outform der -in azure_sp_cert.pem -out azure_sp_cert.cer
cat azure_sp_private_key.pem azure_sp_cert.pem > azure_sp_combined.pem
